85void LoadBalancer::scalingCheckAndApplyIfCooldownPassed() {
86 scaleMessage = "";
87
88 if (currentTick - lastScaleTick < cooldown) {
89 return;
90 }
91
92 int queueSize = q.size();
93 int numServers = ws.size();
94 int thresholdHigh = 25 * numServers;
95 int thresholdLow = 15 * numServers;
96
97 if (queueSize > thresholdHigh) {
98 ws.push_back(WebServer());
99 lastScaleTick = currentTick;
100 serversAddedTicks.push_back(currentTick);
101 scaleMessage = " [SCALE UP to " + std::to_string(ws.size()) + " servers]";
102 }
103 else if (queueSize < thresholdLow && numServers > 1) {
104 for (size_t i = 0; i < ws.size(); i++) {
105 if (ws[i].requestOK()) {
106 ws.erase(ws.begin() + i);
107 lastScaleTick = currentTick;
108 serversRemovedTicks.push_back(currentTick);
109 scaleMessage = " [SCALE DOWN to " + std::to_string(ws.size()) + " servers]";
110 break;
111 }
112 }
113 }
114}
