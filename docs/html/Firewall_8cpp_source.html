27uint32_t Firewall::ipToNumeric(const std::string& ip) {
28 std::istringstream iss(ip);

42void Firewall::loadBlockedRanges(const std::string& filename) {
43 std::ifstream file(filename);
44 if (!file.is_open()) {
45 std::cerr << "Warning: Could not open file " << filename << std::endl;
46 return;
47 }
48
49 std::string line;
50 while (std::getline(file, line)) {
51 // Skip empty lines
52 if (line.empty() || line.find_first_not_of(" \t") == std::string::npos) {
53 continue;
54 }
55
56 // Parse range: format is "startIP-endIP" or "startIP endIP"
57 size_t dashPos = line.find('-');
58 size_t spacePos = line.find(' ');
59
60 if (dashPos != std::string::npos) {
61 // Format: "startIP-endIP"
62 std::string startIP = line.substr(0, dashPos);
63 std::string endIP = line.substr(dashPos + 1);
64
65 // Trim whitespace
66 startIP.erase(0, startIP.find_first_not_of(" \t"));
67 startIP.erase(startIP.find_last_not_of(" \t") + 1);
68 endIP.erase(0, endIP.find_first_not_of(" \t"));
69 endIP.erase(endIP.find_last_not_of(" \t") + 1);
70
71 uint32_t start = ipToNumeric(startIP);
72 uint32_t end = ipToNumeric(endIP);
73
74 blockedRanges.push_back(std::make_pair(start, end));
75 } else if (spacePos != std::string::npos) {
76 // Format: "startIP endIP"
77 std::string startIP = line.substr(0, spacePos);
78 std::string endIP = line.substr(spacePos + 1);
79
80 // Trim whitespace
81 startIP.erase(0, startIP.find_first_not_of(" \t"));
82 startIP.erase(startIP.find_last_not_of(" \t") + 1);
